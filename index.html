function startGame(type){ powerType = type; gamesPlayed++ localStorage.setItem("played",gamesPlayed) played.innerText=gamesPlayed characterMenu.style.display="none" hud.style.display="block" gameActive=true scene=new THREE.Scene() camera=new THREE.PerspectiveCamera( 75, window.innerWidth/window.innerHeight, 0.1, 2000 ) renderer=new THREE.WebGLRenderer({antialias:true}) renderer.setSize(window.innerWidth,window.innerHeight) document.body.appendChild(renderer.domElement) scene.add(new THREE.AmbientLight(0xffffff,1)) let island=new THREE.Mesh( new THREE.CircleGeometry(600,64), new THREE.MeshBasicMaterial({color:0x228B22}) ) island.rotation.x=-Math.PI/2 scene.add(island) let s=skins.find(x=>x.id===currentSkin) let geom if(s.shape==="box")geom=new THREE.BoxGeometry(8,12,8) if(s.shape==="cylinder")geom=new THREE.CylinderGeometry(6,6,12,16) if(s.shape==="cone")geom=new THREE.ConeGeometry(8,16,16) if(s.shape==="sphere")geom=new THREE.SphereGeometry(7,16,16) if(s.shape==="ico")geom=new THREE.IcosahedronGeometry(8) if(s.shape==="oct")geom=new THREE.OctahedronGeometry(8) let mat=new THREE.MeshBasicMaterial({color:s.color||0xffffff}) player=new THREE.Mesh(geom,mat) player.position.y=6 scene.add(player) camera.position.set(0,30,40) camera.lookAt(player.position) animate() }
